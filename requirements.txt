# Ultralytics YOLOv8 - The core object detection model
# Using version 8.3.0 as specified in the README
# YOLOv8 is chosen because:
#   - Extremely fast inference (< 2 seconds on CPU)
#   - Excellent accuracy for road damage detection
#   - Single-line Python API
#   - Transfer learning from COCO dataset (already knows roads/objects)
#   - Built-in support for real-time web deployment
ultralytics==8.3.0

# FastAPI - Modern, fast web framework for building APIs
# Used for the ML inference API endpoint
# Benefits:
#   - Fast performance (similar to Node.js/Go)
#   - Easy to use for Python developers
#   - Automatic API documentation (Swagger UI)
#   - Built-in validation and serialization
fastapi==0.115.0

# Uvicorn - ASGI server for running FastAPI applications
# Required to run the API server
# ASGI (Asynchronous Server Gateway Interface) is the standard
# for building asynchronous web applications in Python

# python-multipart - Multipart form data parsing
